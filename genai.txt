from docx import Document
from sentence_transformers import SentenceTransformer as S
from transformers import pipeline
import faiss,numpy as np
import PyPDF2

t="".join(p.extract_text() or "" for p in PyPDF2.PdfReader(input("PDF: ")).pages)
c=[t[i:i+400] for i in range(0,len(t),400)]

e=S("all-MiniLM-L6-v2"); v=e.encode(c)
f=faiss.IndexFlatL2(v.shape[1]); f.add(np.array(v))

q=input("Q: "); _,I=f.search(e.encode([q]),5)
print(pipeline("text2text-generation",model="google/flan-t5-small")
(f"Use only: {' '.join(c[i] for i in I[0])}\nQ:{q}",max_new_tokens=100)[0]["generated_text"])
